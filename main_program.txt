# SERVOS
# LEFT = servo 0
# RIGHT = servo 1
# MIDDLE= servo 2

# Sequence 0
start:
 # back up and turn left if the right sensor sees an object
  #right_sensor if backward goto start endif
forward
#leftward
goto start

# returns true if the right sensor sees an object
sub right_sensor
 4 get_position 600 greater_than  return
 return

# forward
sub forward
 6800 2 servo # middle, left down
 500 delay
 7000 0 servo # left fwd   
 7000 1 servo #right bwd
 1000 delay

 5200 2 servo # middle, right down
 500 delay
 5000 0 servo # left bwd   
 5000 1 servo #right fwd
 1000 delay  
return

# back
sub backward
 6800 2 servo # middle, left down
 500 delay
 5000 0 servo # left bwd   
 5000 1 servo #right fwd

 1000 delay

 5200 2 servo # middle, right down
 500 delay
 7000 0 servo # left fwd   
 7000 1 servo #right bwd
 1000 delay  
return

# right
sub rightward
 6800 2 servo # middle, left down
 500 delay
 7000  0 servo # left fwd
 5000 1 servo #right fwd
 
 1000 delay

 5200 2 servo # middle, right down
 500 delay
 5000 0 servo # left bwd   
 7000 1 servo #right bwd
 1000 delay  
return

# left
sub leftward
 5200 2 servo # middle, right down
 500 delay
 7000 0 servo # left fwd
 5000 1 servo #right fwd
   
 1000 delay
 
 6800 2 servo # middle, left down
 500 delay
 5000 0 servo # left bwd   
 7000 1 servo #right bwd
 1000 delay  
return
